## POST /v1/auth/register

Register a new user account in the CricPR system. Upon successful registration, the user account is created along with an associated player profile.

### Request Body

<ParamField body="name" type="string" required>
  Full name of the user
</ParamField>

<ParamField body="email" type="string" required>
  Email address (must be unique and valid email format)
</ParamField>

<ParamField body="password" type="string" required>
  Password (minimum 8 characters, must contain at least one number and one letter)
</ParamField>

<ParamField body="role" type="string" required>
  Account role type. Only `user` is accepted for registration.
</ParamField>

<ParamField body="fcmToken" type="string">
  Firebase Cloud Messaging token for push notifications (optional)
</ParamField>

### Response

<ResponseField name="user" type="object">
  The created user object
  
  <ResponseField name="id" type="string">
    User ID
  </ResponseField>
  
  <ResponseField name="name" type="string">
    User's full name
  </ResponseField>
  
  <ResponseField name="email" type="string">
    User's email address
  </ResponseField>
  
  <ResponseField name="role" type="string">
    User's role
  </ResponseField>
  
  <ResponseField name="isEmailVerified" type="boolean">
    Whether the email has been verified
  </ResponseField>
</ResponseField>

<ResponseField name="tokens" type="object">
  Authentication tokens
  
  <ResponseField name="access" type="object">
    Access token details
    
    <ResponseField name="token" type="string">
      JWT access token
    </ResponseField>
    
    <ResponseField name="expires" type="string">
      Token expiration timestamp (ISO 8601)
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="refresh" type="object">
    Refresh token details
    
    <ResponseField name="token" type="string">
      JWT refresh token
    </ResponseField>
    
    <ResponseField name="expires" type="string">
      Token expiration timestamp (ISO 8601)
    </ResponseField>
  </ResponseField>
</ResponseField>

<ResponseField name="playerProfile" type="object">
  The created player profile
  
  <ResponseField name="id" type="string">
    Player profile ID
  </ResponseField>
  
  <ResponseField name="userId" type="string">
    Associated user ID
  </ResponseField>
  
  <ResponseField name="shortName" type="string">
    Player's short name
  </ResponseField>
  
  <ResponseField name="status" type="string">
    Profile status (`active` or `incomplete_profile`)
  </ResponseField>
</ResponseField>

### Example Request

```bash cURL
curl -X POST https://api.cricpr.com/v1/auth/register \
  -H "Content-Type: application/json" \
  -d '{
    "name": "John Smith",
    "email": "john.smith@example.com",
    "password": "password123",
    "role": "user"
  }'
```
