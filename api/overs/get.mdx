---
title: 'Get Overs by Innings'
api: 'GET /v1/overs/inning/{inningId}'
---

Retrieve all overs for a specific innings.

## Path Parameters

<ParamField path="inningId" type="string" required>
  MongoDB ObjectId of the innings
</ParamField>

## Response

<ResponseField name="status" type="string">
  Response status ("success")
</ResponseField>

<ResponseField name="data" type="array">
  Array of over objects
  <ResponseField name="id" type="string">
    Over ID
  </ResponseField>
  <ResponseField name="inningId" type="string">
    Innings ID
  </ResponseField>
  <ResponseField name="bowlerId" type="string">
    Bowler ID
  </ResponseField>
  <ResponseField name="overNumber" type="integer">
    Over number in the innings
  </ResponseField>
  <ResponseField name="runs" type="integer">
    Total runs in this over
  </ResponseField>
  <ResponseField name="wickets" type="integer">
    Wickets taken in this over
  </ResponseField>
  <ResponseField name="balls" type="array">
    Array of ball IDs in this over
  </ResponseField>
  <ResponseField name="isCompleted" type="boolean">
    Whether the over is completed
  </ResponseField>
</ResponseField>

## Example Request

```bash
curl -X GET "https://api.cricpr.com/v1/overs/inning/507f1f77bcf86cd799439011" \
  -H "Content-Type: application/json"
```
