---
title: 'Create Team'
api: 'POST /v1/teams'
---

Create a new team.

## Authentication

This endpoint requires authentication. Include a valid JWT token in the Authorization header.

```
Authorization: Bearer YOUR_TOKEN
```

## Request Body

<ParamField body="name" type="string" required>
  Team name
</ParamField>

<ParamField body="location" type="string" required>
  Team location
</ParamField>

<ParamField body="logo" type="string">
  Team logo URL
</ParamField>

<ParamField body="players" type="array">
  Array of player IDs
</ParamField>

<ParamField body="captain" type="string">
  Captain player ID (must be from the players array)
</ParamField>

<ParamField body="includeMyself" type="boolean">
  Automatically add the authenticated user's player profile to the team
</ParamField>

## Response

<ResponseField name="id" type="string">
  Team ID
</ResponseField>

<ResponseField name="name" type="string">
  Team name
</ResponseField>

<ResponseField name="location" type="string">
  Team location
</ResponseField>

<ResponseField name="logo" type="string">
  Team logo URL
</ResponseField>

<ResponseField name="captain" type="object">
  Captain player details
</ResponseField>

<ResponseField name="players" type="array">
  Array of player objects
</ResponseField>

<ResponseField name="createdBy" type="string">
  Player ID of the creator
</ResponseField>
