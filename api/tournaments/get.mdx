---
title: 'Get Tournament'
api: 'GET /v1/tournaments/:tournamentId'
---

Retrieve details of a specific tournament.

## Path Parameters

<ParamField path="tournamentId" type="string" required>
  The ID of the tournament to retrieve
</ParamField>

## Response

<ResponseField name="id" type="string">
  Tournament ID
</ResponseField>

<ResponseField name="name" type="string">
  Tournament name
</ResponseField>

<ResponseField name="coverImage" type="string">
  Cover image URL
</ResponseField>

<ResponseField name="banner" type="string">
  Banner URL
</ResponseField>

<ResponseField name="city" type="string">
  City
</ResponseField>

<ResponseField name="startDate" type="date">
  Start date
</ResponseField>

<ResponseField name="endDate" type="date">
  End date
</ResponseField>

<ResponseField name="maxTeams" type="integer">
  Maximum teams allowed
</ResponseField>

<ResponseField name="format" type="string">
  Tournament format
</ResponseField>

<ResponseField name="status" type="string">
  Current status (upcoming, ongoing, completed, cancelled)
</ResponseField>

<ResponseField name="organizerId" type="object">
  Organizer details
</ResponseField>

<ResponseField name="organizerType" type="string">
  Organizer type
</ResponseField>

<ResponseField name="description" type="string">
  Tournament description
</ResponseField>

<ResponseField name="registrationDeadline" type="date">
  Registration deadline
</ResponseField>

<ResponseField name="prize" type="string">
  Prize details
</ResponseField>

<ResponseField name="entryFee" type="number">
  Entry fee
</ResponseField>

<ResponseField name="isPublic" type="boolean">
  Public visibility
</ResponseField>
