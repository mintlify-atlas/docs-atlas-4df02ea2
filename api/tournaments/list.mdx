---
title: 'List Tournaments'
api: 'GET /v1/tournaments'
---

Retrieve a paginated list of tournaments with optional filters.

## Authentication

This endpoint requires authentication. Include a valid JWT token in the Authorization header.

```
Authorization: Bearer YOUR_TOKEN
```

## Query Parameters

<ParamField query="organizerId" type="string">
  Filter by organizer ID
</ParamField>

<ParamField query="organizerType" type="string">
  Filter by organizer type. One of: `user`, `player`
</ParamField>

<ParamField query="city" type="string">
  Filter by city
</ParamField>

<ParamField query="status" type="string">
  Filter by status. One of: `upcoming`, `ongoing`, `completed`, `cancelled`
</ParamField>

<ParamField query="format" type="string">
  Filter by format. One of: `round_robin`, `knockout`, `hybrid`, `point_based`
</ParamField>

<ParamField query="isPublic" type="boolean">
  Filter by public visibility
</ParamField>

<ParamField query="participant" type="boolean">
  Filter tournaments where the authenticated user is a participant (legacy parameter)
</ParamField>

<ParamField query="participated" type="boolean">
  Filter tournaments where the authenticated user has participated (default: false)
</ParamField>

<ParamField query="past" type="boolean">
  Include past matches in the results
</ParamField>

<ParamField query="sortBy" type="string">
  Sort field in format `field:(asc|desc)`. Default: `createdAt:desc`
</ParamField>

<ParamField query="limit" type="integer">
  Number of items per page
</ParamField>

<ParamField query="page" type="integer">
  Page number
</ParamField>

## Response

<ResponseField name="results" type="array">
  Array of tournament objects
</ResponseField>

<ResponseField name="page" type="integer">
  Current page number
</ResponseField>

<ResponseField name="limit" type="integer">
  Items per page
</ResponseField>

<ResponseField name="totalPages" type="integer">
  Total number of pages
</ResponseField>

<ResponseField name="totalResults" type="integer">
  Total number of tournaments
</ResponseField>
