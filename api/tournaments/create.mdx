---
title: 'Create Tournament'
api: 'POST /v1/tournaments'
---

Create a new tournament with specified details.

## Authentication

This endpoint requires authentication. Include a valid JWT token in the Authorization header.

```
Authorization: Bearer YOUR_TOKEN
```

## Request Body

<ParamField body="name" type="string" required>
  Tournament name
</ParamField>

<ParamField body="city" type="string" required>
  City where the tournament will be held
</ParamField>

<ParamField body="startDate" type="date" required>
  Tournament start date
</ParamField>

<ParamField body="endDate" type="date" required>
  Tournament end date (must be after startDate)
</ParamField>

<ParamField body="maxTeams" type="integer" required>
  Maximum number of teams allowed (minimum: 2)
</ParamField>

<ParamField body="format" type="string" required>
  Tournament format. One of: `round_robin`, `knockout`, `hybrid`, `point_based`
</ParamField>

<ParamField body="organizerType" type="string" required>
  Type of organizer. One of: `User`, `Player`
</ParamField>

<ParamField body="organizerId" type="string" required>
  ID of the organizer (User or Player ID based on organizerType)
</ParamField>

<ParamField body="coverImage" type="string">
  URL of the tournament cover image
</ParamField>

<ParamField body="banner" type="string">
  URL of the tournament banner
</ParamField>

<ParamField body="description" type="string">
  Tournament description
</ParamField>

<ParamField body="registrationDeadline" type="date">
  Deadline for team registration
</ParamField>

<ParamField body="prize" type="string">
  Prize details
</ParamField>

<ParamField body="entryFee" type="number">
  Entry fee amount (minimum: 0, default: 0)
</ParamField>

<ParamField body="isPublic" type="boolean">
  Whether the tournament is public (default: true)
</ParamField>

</ResponseField>

  Organizer ID
</ResponseField>

  Organizer type
</ResponseField>
